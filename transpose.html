Matrix transposition

      m = Amj - Ami, n = Anj - Ani
      Bmj = Bmi + n, Bnj = Bni + m

      A                          B

Ani Anj Bni Bnj Ami 1 2 . . . . .. Bmi 1 3 . . . .. .. 3 . . . . . .. .. 2 . . . . .. .. . . . . . . .. -> .. . . . . . .. .. . . . . . . .. .. . . . . . .. .. . . . . . . .. .. . . . . . .. Amj .. . . . . . .. Bmj

Parameters:

  • A must be a matrix.
    (matrix pointer)

  • Ami must be a const index.

  • Amj must be a const index.

  • Ani must be a const index.

  • Anj must be a const index.

  • B must be a matrix.
    (matrix pointer)

  • Bmi must be a const index.

  • Bni must be a const index.

var transpose = function ( A, Ami, Amj, Ani, Anj, B, Bmi, Bni ) { var m, n, r, c, Ar, Br; m = Amj - Ami; n = Anj - Ani; for ( r = 0 ; r < m ; ++r ) { Ar = A[Ami + r]; for ( c = 0 ; c < n ; ++c ) { B[Bmi + c][Bni + r] = Ar[Ani + c]; } } }; exports.transpose = transpose;
