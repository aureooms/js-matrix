Matrix copy

      m = Amj - Ami, n = Anj - Ani
      Bmj = Bmi + m, Bnj = Bni + n

      A                           B

Ani Anj Bni Bnj Ami . . . . . . .. Bmi . . . . . . .. .. . . . . . . .. .. . . . . . . .. .. . . . . . . .. -> .. . . . . . . .. .. . . . . . . .. .. . . . . . . .. .. . . . . . . .. .. . . . . . . .. Amj Bmj

Parameters:

  • A must be a matrix.
    (matrix pointer)

  • Ami must be a const index.

  • Amj must be a const index.

  • Ani must be a const index.

  • Anj must be a const index.

  • B must be a matrix.
    (matrix pointer)

  • Bmi must be a const index.

  • Bni must be a const index.

var copy = function ( A, Ami, Amj, Ani, Anj, B, Bmi, Bni ) { var m, n, r, c, Ar, Br; m = Amj - Ami; n = Anj - Ani; for ( r = 0 ; r < m ; ++r ) { Ar = A[Ami + r]; Br = B[Bmi + r]; for ( c = 0 ; c < n ; ++c ) { Br[Bni + c] = Ar[Ani + c]; } } }; exports.copy = copy;
